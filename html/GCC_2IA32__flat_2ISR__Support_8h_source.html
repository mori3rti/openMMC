77  /* Save general purpose registers. */
78  pusha

80  /* If ulInterruptNesting is zero the rest of the task context will need
81  saving and a stack switch might be required. */
82  movl ulInterruptNesting, %eax
83  test %eax, %eax
84  jne 2f
85 
86  /* Interrupts are not nested, so save the rest of the task context. */
87  .if configSUPPORT_FPU == 1
88 
89  /* If the task has a buffer allocated to save the FPU context then
90  save the FPU context now. */
91  movl pucPortTaskFPUContextBuffer, %eax

95  fwait
