119 { \
120 extern volatile /*uint16_t*/ portSTACK_TYPE usCriticalNesting; \

123  \
124  /* Now interrupts are disabled ulCriticalNesting can be accessed */ \

130 #define portEXIT_CRITICAL() \
131 { \
132 extern volatile /*uint16_t*/ portSTACK_TYPE usCriticalNesting; \
133  \
134  if( usCriticalNesting > portNO_CRITICAL_SECTION_NESTING ) \
135  { \
136  /* Decrement the nesting count as we are leaving a critical section. */ \
137  usCriticalNesting--; \
138  \
139  /* If the nesting level has reached zero then interrupts should be */ \
140  /* re-enabled. */ \
141  if( usCriticalNesting == portNO_CRITICAL_SECTION_NESTING ) \
142  { \

144  } \
