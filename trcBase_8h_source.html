00081 /******************************************************************************
00082  * ObjectHandleStack
00083  * This data-structure is used to provide a mechanism for 1-byte trace object
00084  * handles. This way, only 1 byte is necessary instead of 4 bytes (a pointer)
00085  * when storing a reference to an object. This allows for up to 255 objects of
00086  * each object class active at any given moment. There can be more "historic"
00087  * objects, that have been deleted - that number is only limited by the size of
00088  * the symbol table.
00089  * Note that handle zero (0) is not used, it is a code for an invalid handle.
00090  *
00091  * This data structure keeps track of the FREE handles, not the handles in use.
00092  * This data structure contains one stack per object class. When a handle is
00093  * allocated to an object, the next free handle is popped from the stack. When
00094  * a handle is released (on object delete), it is pushed back on the stack.
00095  * Note that there is no initialization code that pushed the free handles
00096  * initially, that is not necessary due to the following optimization:
00097  *
00098  * The stack of handles (objectHandles) is initially all zeros. Since zero
00099  * is not a valid handle, that is a signal of additional handles needed.
00100  * If a zero is received when popping a new handle, it is replaced by the
00101  * index of the popped handle instead.
00102  *
00103  *****************************************************************************/
00104 typedef struct
00105 {
00106     /* For each object class, the index of the next handle to allocate */
00107     uint16_t indexOfNextAvailableHandle[ TRACE_NCLASSES ];
00108 
00109     /* The lowest index of this class (constant) */
00110     uint16_t lowestIndexOfClass[ TRACE_NCLASSES ];
00111 
00112     /* The highest index of this class (constant) */
00113     uint16_t highestIndexOfClass[ TRACE_NCLASSES ];
00114 
00115     /* The highest use count for this class (for statistics) */
00116     uint16_t handleCountWaterMarksOfClass[ TRACE_NCLASSES ];
00117 
00118     /* The free object handles - a set of stacks within this array */
00119     objectHandleType objectHandles[ TRACE_KERNEL_OBJECT_COUNT ];
00120 
00121 } objectHandleStackType;

00126  * Object Property Table
00127  * The Object Table contains name and other properties of the objects (tasks,
00128  * queues, mutexes, etc). The below data structures defines the properties of
00129  * each object class and are used to cast the byte buffer into a cleaner format.
00130  *
00131  * The values in the object table are continuously overwritten and always
00132  * represent the current state. If a property is changed during runtime, the OLD
00133  * value should be stored in the trace buffer, not the new value (since the new
00134  * value is found in the Object Property Table).
00135  * For close events this mechanism is the old names are stored in the symbol
00136  * table), for "priority set" (the old priority is stored in the event data)
00137  * and for "isActive", where the value decides if the task switch event type
00138  * should be "new" or "resume".
00139  ******************************************************************************/
